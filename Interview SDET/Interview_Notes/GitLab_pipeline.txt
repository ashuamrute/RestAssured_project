stages:
  - build
  - test
  - deploy

# Build stage...........
build-job:
  stage: build
  script:
    - echo "Compiling the project..."
    - mvn clean install
  artifacts:
    paths:
      - target/*.jar

# Test stage...........
test-job:
  stage: test
  script:
    - echo "Running tests..."
    - mvn test

# Deploy stage............
deploy-job:
  stage: deploy
  script:
    - echo "Deploying application..."
    - scp target/*.jar user@server:/deployments/
  only:
    - main
